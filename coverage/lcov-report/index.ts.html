function isPromise2<TVal, TErr>(maybePromise: TVal | TErr | Promise2<TVal, TErr> | unknown): maybePromise is Promise2<TVal, TErr> {

function isThenable<T>(maybePromise: PromiseLike<T> | unknown): maybePromise is PromiseLike<T> {
